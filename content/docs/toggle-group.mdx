## Installation

import { Step, Steps } from 'fumadocs-ui/components/steps';

<Steps>
<Step>
Install the Toggle Group component using the nocta-ui CLI:

```bash lineNumbers title="Terminal"
npx @nocta-ui/cli add toggle-group
```

</Step>
<Step>
Import the components:

```tsx lineNumbers title="Import"
import {
  ToggleGroup,
  ToggleGroupItem
} from '@/components/ui/toggle-group';
```

</Step>
</Steps>

### Sizes

<DocsTabs defaultValue="preview">

      <ToggleGroupSizesDemo />
    </div>
  </DocsTab>
  <DocsTab value="code" lineNumbers title="Code">
    ```tsx lineNumbers title="Toggle Group Sizes"
    const [values, setValues] = useState({ sm: ['left'], md: ['center'], lg: ['right'] });

    <ToggleGroup single size="sm" value={values.sm} onValueChange={(next) => setValues(prev => ({ ...prev, sm: next }))}>
      <ToggleGroupItem value="left">Left</ToggleGroupItem>
      <ToggleGroupItem value="center">Center</ToggleGroupItem>
      <ToggleGroupItem value="right">Right</ToggleGroupItem>
    </ToggleGroup>
    <ToggleGroup single size="md" value={values.md} onValueChange={(next) => setValues(prev => ({ ...prev, md: next }))}>
      <ToggleGroupItem value="left">Left</ToggleGroupItem>
      <ToggleGroupItem value="center">Center</ToggleGroupItem>
      <ToggleGroupItem value="right">Right</ToggleGroupItem>
    </ToggleGroup>
    <ToggleGroup single size="lg" value={values.lg} onValueChange={(next) => setValues(prev => ({ ...prev, lg: next }))}>
      <ToggleGroupItem value="left">Left</ToggleGroupItem>
      <ToggleGroupItem value="center">Center</ToggleGroupItem>
      <ToggleGroupItem value="right">Right</ToggleGroupItem>
    </ToggleGroup>
    ```
