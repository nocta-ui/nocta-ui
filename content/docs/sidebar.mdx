## Installation

<Steps>
  <Step>
    Install the Sidebar package using the nocta-ui CLI:

```bash lineNumbers title="Terminal"
npx @nocta-ui/cli add sidebar
```
  </Step>
  <Step>
    Then import the primitives you need:

```tsx lineNumbers title="Import"
import {
  Sidebar,
  SidebarContent,
  SidebarFooter,
  SidebarGroup,
  SidebarGroupAction,
  SidebarGroupContent,
  SidebarGroupLabel,
  SidebarHeader,
  SidebarInset,

```
  </Step>
</Steps>

```tsx lineNumbers title="Sidebar layout"
import {
  SidebarProvider,
  Sidebar,
  SidebarHeader,
  SidebarInput,
  SidebarContent,
  SidebarGroup,
  SidebarGroupLabel,
  SidebarGroupContent,
  SidebarMenu,
  SidebarMenuItem,
  SidebarMenuButton,
  SidebarFooter,
  SidebarTrigger,
  SidebarInset,
} from '@/components/ui/sidebar';

export function AppSidebar() {
  return (
    <SidebarProvider className="min-h-[28rem] rounded-2xl border border-border/70 bg-background/80 shadow-card">
      <Sidebar>
        <SidebarHeader>
          <SidebarInput placeholder="Search workspace" />
        </SidebarHeader>
        <SidebarContent>
          <SidebarGroup>
            <SidebarGroupLabel>Workspace</SidebarGroupLabel>
            <SidebarGroupContent>
              <SidebarMenu>
                <SidebarMenuItem>
                  <SidebarMenuButton isActive>
                    <span>Overview</span>
                  </SidebarMenuButton>
                </SidebarMenuItem>
                <SidebarMenuItem>
                  <SidebarMenuButton>
                    <span>Launch plans</span>
                  </SidebarMenuButton>
                </SidebarMenuItem>
              </SidebarMenu>
            </SidebarGroupContent>
          </SidebarGroup>
        </SidebarContent>
        <SidebarFooter>
          <p className="text-sm text-foreground/70">
            Hand off work or show reminders here.
          </p>
        </SidebarFooter>
      </Sidebar>
      <SidebarTrigger />
      <SidebarInset className="p-6 space-y-4">
        <div>
          <p className="text-xs text-foreground/60">Active sprint</p>
          <p className="text-2xl font-semibold text-foreground">Northstar dashboard</p>
        </div>
        <div className="rounded-xl border border-border/60 bg-card/70 p-4 shadow-card">
          Keep your primary content inside the inset. It automatically grows to
          fill the space when the sidebar collapses.
        </div>
      </SidebarInset>
    </SidebarProvider>
  );
}
```
