## Installation

<Steps>
  <Step>
    Install the Command Palette component using the nocta-ui CLI:

    ```bash lineNumbers title="Terminal"
    npx nocta-ui add command-k
    ```
  </Step>
  <Step>
    Import the component:

    ```tsx lineNumbers title="Import"
    import { CommandK } from '@/components/ui/command-k';
    ```
  </Step>
</Steps>

## Example

### Default Command Palette

<DocsTabs defaultValue="preview">
  <DocsTab value="preview" lineNumbers title="Preview">
    <BasicCommandKDemo />
  </DocsTab>
  <DocsTab value="code" lineNumbers title="Code">
    ```tsx lineNumbers title="Basic Command Palette"
     const items: CommandKItem[] = [
		{
			label: "Search docs",
			group: "Navigation",
			description: "Go to documentation",
			icon: IconSearch,
			shortcut: ["S"],
		},
		{
			label: "Open settings",
			group: "Navigation",
			description: "Application preferences",
			icon: IconCog,
			shortcut: ["O"],
		},
		{
			label: "New file",
			group: "Actions",
			description: "Create an untitled file",
			icon: IconFile,
			shortcut: ["Shift", "F"],
		},
		{
			label: "New folder",
			group: "Actions",
			description: "Create a folder",
			icon: IconFolder,
			shortcut: ["Shift", "G"],
		},
	];

  const [open, setOpen] = useState(false);

  <button onClick={() => setOpen(true)}>Open Command Palette</button>
  <CommandK
      items={items}
      open={open}
      onOpenChange={setOpen}
      onSelect={(it) => {
      toast.success({
        title: "Command executed",
        description: `Selected: ${it.label}`,
      });
    }}
  />;
    ```
  </DocsTab>
</DocsTabs>
