## Installation

<Steps>
  <Step>
    Install the OTP Input component using the nocta-ui CLI:

    ```bash lineNumbers title="Terminal"
    npx @nocta-ui/cli add otp-input
    ```
  </Step>
  <Step>
    Import the component:

    ```tsx lineNumbers title="Import"
    import { OTPInput } from '@/components/ui/otp-input';
    ```
  </Step>
</Steps>

  <DocsTab value="code" lineNumbers title="Code">
```tsx lineNumbers title="OTP verification flow"
const [code, setCode] = useState('');
const isComplete = code.length === 6;

return (
  <OTPInput
    length={6}
    value={code}
    onChange={setCode}
    variant={isComplete ? 'success' : 'default'}
    label="One-time password"
    helperText="Enter the 6-digit code sent to your phone."
    successMessage={isComplete ? 'Code captured successfully.' : undefined}
    autoComplete="one-time-code"
  />
);
```
