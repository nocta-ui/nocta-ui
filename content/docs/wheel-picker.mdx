## Installation

<Steps>
<Step>
Install the Wheel Picker via the nocta-ui CLI:

```bash lineNumbers title="Terminal"
npx @nocta-ui/cli add wheel-picker
```
</Step>
<Step>
Import the components you need:

```tsx lineNumbers title="Import"
import {
  WheelPicker,
  WheelPickerGroup,
  WheelPickerItem,
} from '@/components/ui/wheel-picker';
```
</Step>
</Steps>

### Grouped

<DocsTabs defaultValue="preview">
  <DocsTab value="preview" lineNumbers title="Preview">
    <GroupedWheelPickerDemo />
  </DocsTab>
  <DocsTab value="code" lineNumbers title="Code">
    ```tsx lineNumbers title="Grouped"
    const months = [
      'January','February','March','April','May','June',
      'July','August','September','October','November','December',
    ];
    const days = Array.from({ length: 31 }, (_, i) => (i + 1).toString().padStart(2, '0'));
    const years = Array.from({ length: 11 }, (_, i) => (2020 + i).toString());
    const [month, setMonth] = React.useState(months[5]);
    const [day, setDay] = React.useState(days[9]);
    const [year, setYear] = React.useState(years[4]);

    <WheelPickerGroup>
      <WheelPicker className="w-28" value={month} onValueChange={setMonth} aria-label="Month">
        {months.map((m) => (
          <WheelPickerItem key={m} value={m}>
            {m}
          </WheelPickerItem>
        ))}
      </WheelPicker>

      <WheelPicker className="w-10" value={day} onValueChange={setDay} aria-label="Day">
        {days.map((d) => (
          <WheelPickerItem key={d} value={d}>
            {d}
          </WheelPickerItem>
        ))}
      </WheelPicker>

      <WheelPicker className="w-24" value={year} onValueChange={setYear} aria-label="Year">
        {years.map((y) => (
          <WheelPickerItem key={y} value={y}>
            {y}
          </WheelPickerItem>
        ))}
      </WheelPicker>
    </WheelPickerGroup>
    ```
  </DocsTab>
</DocsTabs>
