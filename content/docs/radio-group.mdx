## Installation

<Steps>
  <Step>
    Install the Radio Group component using the nocta-ui CLI:

    ```bash lineNumbers title="Terminal"
    npx nocta-ui add radio-group
    ```
  </Step>
  <Step>
    Import the component:

    ```tsx lineNumbers title="Import"
    import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group';
    ```
  </Step>
</Steps>

    ```tsx lineNumbers title="Basic Radio Group"
    const [theme, setTheme] = useState('system');

    <RadioGroup
      value={theme}
      onValueChange={(v) => setTheme(v as string)}
      aria-label="Theme"
    >
      <RadioGroupItem value="light" label="Light" />
      <RadioGroupItem value="dark" label="Dark" />
      <RadioGroupItem value="system" label="System" />
    </RadioGroup>
    ```
