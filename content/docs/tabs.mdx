## Installation

import { Step, Steps } from 'fumadocs-ui/components/steps';

<Steps>
<Step>
Install the Tabs component using the nocta-ui CLI:

```bash lineNumbers title="Terminal"
npx nocta-ui add tabs
```
</Step>
<Step>
Import the component:

```tsx lineNumbers title="Import"
import {
  Tabs,
  TabsList,
  TabsTrigger,
  TabsContent
} from '@/components/ui/tabs';
```
</Step>
</Steps>

### Vertical Orientation

Tabs can be displayed vertically using the `orientation="vertical"` prop.

<DocsTabs defaultValue="preview">
  <DocsTab value="preview" lineNumbers title="Preview">
    <VerticalTabsDemo />
  </DocsTab>
  <DocsTab value="code" lineNumbers title="Code">
    ```tsx lineNumbers title="Vertical Tabs"
    <Tabs defaultValue="general" orientation="vertical">
      <TabsList>
        <TabsTrigger value="general">General</TabsTrigger>
        <TabsTrigger value="security">Security</TabsTrigger>
        <TabsTrigger value="integrations">Integrations</TabsTrigger>
        <TabsTrigger value="advanced">Advanced</TabsTrigger>
      </TabsList>
      <div className="flex-1">
        <TabsContent value="general">
          <Card>
            <CardHeader>
              <CardTitle>General Settings</CardTitle>
              <CardDescription>
                Manage your general account settings.
              </CardDescription>
            </CardHeader>
            <CardContent>
              <p>General settings content.</p>
            </CardContent>
          </Card>
        </TabsContent>
        {/* Other TabsContent components */}
      </div>
    </Tabs>
    ```
  </DocsTab>
</DocsTabs>
