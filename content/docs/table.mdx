## Installation

<Steps>
  <Step>
    Install the Table component using the nocta-ui CLI:

    ```bash lineNumbers title="Terminal"
    npx @nocta-ui/cli add table
    ```
  </Step>
  <Step>
    Then import the component and supporting types:

    ```tsx lineNumbers title="Import"
    import { Table, type ColumnDef } from '@/components/ui/table';
    ```
  </Step>
</Steps>

    interface Invoice {
      invoice: string;
      customer: string;
      status: 'paid' | 'processing' | 'overdue';
      issuedOn: Date;
      amount: number;
    }

    const currency = new Intl.NumberFormat('en-US', {
      style: 'currency',
      currency: 'USD',
      minimumFractionDigits: 0,
    });

    const columns: ColumnDef<Invoice>[] = [
      {
        accessorKey: 'invoice',
        header: 'Invoice',
        cell: ({ row }) => (
          <span className="font-medium">{row.getValue('invoice')}</span>
        ),
        meta: { width: '140px' },
      },
      {
        accessorKey: 'customer',
        header: 'Customer',
      },
      {
        accessorKey: 'issuedOn',
        header: 'Issued',
        cell: ({ getValue }) => {
          const date = getValue<Date>();
          return date.toLocaleDateString('en-US', {
            month: 'short',
            day: 'numeric',
            year: 'numeric',
          });
        },
        meta: { align: 'right', width: '160px' },
      },
      {
        accessorKey: 'amount',
        header: 'Amount',
        cell: ({ getValue }) => (
          <span className="font-medium">
            {currency.format(getValue<number>())}
          </span>
        ),
        meta: { align: 'right', width: '140px' },
      },
    ];

    const data: Invoice[] = [
      {
        invoice: 'INV-2041',
        customer: 'Lumen Studio',
        status: 'paid',
        issuedOn: new Date('2024-02-14'),
        amount: 2450,
      },
      // ...
    ];

    <Table<Invoice>
      columns={invoiceColumns}
      data={invoiceData}
      getRowId={(invoice) => invoice.invoice}
      density="comfortable"
	/>
