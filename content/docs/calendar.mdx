## Installation

<Steps>
  <Step>
    Install the Calendar component using the nocta-ui CLI:

    ```bash lineNumbers title="Terminal"
    npx @nocta-ui/cli add calendar
    ```
  </Step>
  <Step>
    Then import the component:

    ```tsx lineNumbers title="Import"
    import { Calendar } from '@/components/ui/calendar';
    ```
  </Step>
</Steps>

### Custom Formatting

<DocsTabs defaultValue="preview">
  <DocsTab value="preview" lineNumbers title="Preview">
    <CustomFormattingDemo />
  </DocsTab>
  <DocsTab value="code" lineNumbers title="Code">
    ```tsx lineNumbers title="Custom Formatting"
    const formatMonth = (date: Date) => {
      const month = date.toLocaleDateString('en-US', { month: 'long' });
      const year = date.getFullYear();
      return `${month} ${year}`;
    };

    const formatWeekday = (date: Date) => {
      return date.toLocaleDateString('en-US', { weekday: 'short' }).toUpperCase();
    };

    <Calendar
      formatMonth={formatMonth}
      formatWeekday={formatWeekday}
    />
    ```
  </DocsTab>
</DocsTabs>

    ```tsx
    const [selectedDate, setSelectedDate] = useState<Date | undefined>();

    const handleDateSelect = (date: Date | undefined) => {
      setSelectedDate(date);
      if (date) {
        console.log('Selected date:', date.toLocaleDateString());
      }
    };

    <Calendar
      variant="compact"
      value={selectedDate}
      onChange={handleDateSelect}
    />
    {selectedDate && (
      <div className="mt-4 p-3 bg-card rounded-lg">
        <p className="text-sm">
          Selected: <span className="font-medium">{selectedDate.toLocaleDateString()}</span>
        </p>
      </div>
    )}
    ```
