## Installation

import { Step, Steps } from 'fumadocs-ui/components/steps';

<Steps>
<Step>
Install the Toast component using the nocta-ui CLI:

```bash lineNumbers title="Terminal"
npx @nocta-ui/cli add toast
```
</Step>
<Step>
Import the components you need:

```tsx lineNumbers title="Import"
import { toast, Toaster } from '@/components/ui/toast';
```
</Step>
<Step>
Add the Toaster component to your app to enable toast notifications:

```tsx lineNumbers title="Toaster Setup"
import { Toaster } from '@/components/ui/toast';

function App() {
  return (
    <>
      <YourApp />
      <Toaster />
    </>
  );
}
```
</Step>
</Steps>

        });

          title: 'Third Toast',
          description: 'This is the third notification.',
        });
      }, 1000);

        toast.error({
          title: 'Fourth Toast',
          description: 'This will push the first one out!',

      }, 1500);
    };
