Install the Line Chart component using the nocta-ui CLI:

```bash lineNumbers title="Terminal"
npx @nocta-ui/cli add line-chart
```
</Step>
<Step>
Import the component:

```tsx lineNumbers title="Import"
import {
  LineChart,
  LineChartHeader,
  LineChartTitle,
  LineChartDescription,
  LineChartActions,
  LineChartGraph,
  LineChartLegend,
```

</Step>
</Steps>

## API Reference

### LineChart

<TypeTable
  type={{
    data: {
      description: 'Primary series consumed by the chart.',
      type: 'TData[]',
      default: '[]',
    },
    xKey: {
      description: 'Property used for the horizontal axis.',
      type: 'Extract<keyof TData, string>',
      default: '—',
    },
    yKey: {
      description: 'Property that contains the numeric value for each point.',
      type: 'Extract<keyof TData, string>',
      default: '—',
    },
    nameKey: {
      description: 'Optional label key used in tooltips and legends.',
      type: 'Extract<keyof TData, string>',
      default: 'undefined',
    },
    color: {
      description: 'Override the primary series color.',
      type: 'string',
      default: 'Design tokens',
    },
    seriesLabel: {
      description: 'Readable label for the primary series.',
      type: 'string',
      default: 'Derived from nameKey or yKey',
    },
    sortPoints: {
      description: 'Control point ordering or supply a custom comparator.',
      type: 'boolean | ((a: TData, b: TData) => number)',
      default: 'true',
    },
    xFormatter: {
      description: 'Custom formatter for x-axis labels and tooltips.',
      type: '(point: LineChartPoint<TData>) => React.ReactNode',
      default: 'point.label',
    },
    yFormatter: {
      description: 'Formatter for values shown in tooltips and tables.',
      type: '(point: LineChartPoint<TData>, meta: { total: number; domain: [number, number] }) => React.ReactNode',
      default: 'point.value.toLocaleString()',
    },
    valueFormatter: {
      description: 'Overrides the numeric formatter when rendering values.',
      type: '(point: LineChartPoint<TData>, meta: { total: number; domain: [number, number] }) => React.ReactNode',
      default: 'point.value.toLocaleString()',
    },
    additionalSeries: {
      description: 'Render extra series alongside the primary dataset.',
      type: 'LineChartAdditionalSeries<TData>[]',
      default: '[]',
    },
    children: {
      description: 'Compose headers, graphs, legends, and data tables.',
      type: 'React.ReactNode',
      default: 'undefined',
    },
  }}
/>
